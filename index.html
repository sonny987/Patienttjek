// OUH Active Directory SSO via SAML 2.0
import { getAuth, SAMLAuthProvider,
  signInWithPopup } from "firebase/auth";
import { getFirestore, doc, setDoc,
  getDocs, collection, where, query,
  serverTimestamp } from "firebase/firestore";

const provider = new SAMLAuthProvider(
  "saml.ouh-active-directory"
);
async function loginMedAD() {
  const result = await signInWithPopup(
    getAuth(), provider);
  const u = result.user;
  await setDoc(doc(db,"brugere",u.uid),{
    navn: u.displayName, email: u.email,
    afdeling, rolle,
    sidstLogin: serverTimestamp()
  },{ merge:true });
}
// Firestore regler:
match /nyheder/{id} {
  allow read: if request.auth != null
    && get(/brugere/$(request.auth.uid))
       .data.afdeling == resource.data.afd;
  allow write,delete: if
    get(/brugere/$(request.auth.uid))
    .data.rolle == "leder";
}
match /brugere/{uid}/laestNyheder/{id}{
  allow read,write:
    if request.auth.uid == uid;
}
